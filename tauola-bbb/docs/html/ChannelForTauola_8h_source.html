151inline void ChannelForTauola::print() const {
152
153 printf("Channel %3i: '%-31s' b.ratio: %6.5f, multipl-1: %i, subch: %2i ME-type: %i Final state:",
154 m_channel_no,
155 m_name.c_str(),
156 m_br,
157 m_mulpik,
158 m_subchannel_no,
159 m_me_type);
160
161 for(int i=0; i<9; ++i) {
162 int precision = 6;
163 if( i > 3 ) precision = 3;
164
165 if( i == m_mulpik ) printf("|");
166 else printf(" ");
167
168 if( i < (int)m_products.size() ) printf("%*i",precision,m_products[i]);
169 else printf("%*i",precision,0);
170 }
171
172 if( m_me_type > 3 ) {
173 if( m_pointer == NULL ) {
174 printf("\n ERROR: me type is %i and pointer to user function not set!",m_me_type);
175 exit(-1);
176 }
177 else printf("\n Pointer to user function set to: %p",m_pointer);
178 }
179
180 printf("\n");
181}
